	package main
	
	import (
		"io"
		"os"
		"strings"
	)
	
	type rot13Reader struct {
		r io.Reader
	}
	
	func (p *rot13Reader)Read(c []byte)(i int,e error){
		
		i,e = p.r.Read(c)
		for i:=range c {
			switch {
			case c[i]<'N':c[i]+=13
			case c[i]<'a':c[i]-=13
			case c[i]<'n':c[i]+=13
			case c[i]<='z':c[i]-=13
			default:
			}
		}
		return i,e
	}
	
	
	
	func main() {
		s := strings.NewReader("Lbh penpxrq gur pbqr!")
		r := rot13Reader{s}
		io.Copy(os.Stdout, &r)
	}
